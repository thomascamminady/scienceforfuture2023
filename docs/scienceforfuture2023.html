Sortieralgorithmus

import random


def algorithmus(zahlen):
    while not ist_sortiert(zahlen):
        random.shuffle(zahlen)
    return zahlen


def ist_sortiert(zahlen):
    for i in range(len(zahlen) - 1):
        if zahlen[i] > zahlen[i + 1]:
            return False
    return True
